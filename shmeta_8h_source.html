00033 /**
00034  * A meta definition is part of a @c shmeta_t hashmap.
00035  *
00036  * The share library meta definitions can be used to hash header information from a socket stream, retaining access to the meta information by a token, and allowing for efficient redelivery or caching.
00037  *
00038  * In reference to the internal form of a hash table:
00039  *
00040  * The table is an array indexed by the hash of the key; collisions
00041  * are resolved by hanging a linked list of hash entries off each
00042  * element of the array. Although this is a really simple design it
00043  * isn't too bad given that pools have a low allocation overhead.
00044  *
00045  *  @note A @c shfs_tree sharefs file system associates meta definition information with every @c shfs_node inode entry.
00046  *  @ingroup libshare
00047  * @defgroup libshare_meta Meta definition hash maps (meta maps).
00048  * @{
00049  */
00050

00125 /**
00126  * Data structure for iterating through a hash table.
00127  *
00128  * We keep a pointer to the next hash entry here to allow the current
00129  * hash entry to be freed or otherwise mangled between calls to
00130  * shmeta_next().
00131  */
00132 struct shmeta_index_t {
00133     shmeta_t         *ht;
00134     shmeta_entry_t   *this, *next;
00135     unsigned int        index;
00136 };
00137 
00138 /**
00139  * The size of the array is always a power of two. We use the maximum
00140  * index rather than the size so that we can use bitwise-AND for
00141  * modular arithmetic.
00142  * The count of hash entries may be greater depending on the chosen
00143  * collision rate.
00144  */
00145 struct shmeta_t {
00146     shmeta_entry_t   **array;
00147     shmeta_index_t     iterator;  /* For shmeta_first(...) */
00148     unsigned int         count, max;
00149     shmetafunc_t       hash_func;
00150     shmeta_entry_t    *free;  /* List of recycled entries */
00151 };
