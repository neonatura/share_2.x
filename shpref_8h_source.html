00067 /**
00068  * Specifies whether to track sharefs filesystem revisions.
00069  * @returns A zero (0) when disabled and a non-zero value when enabled.
00070  */
00071 #define shpref_track() \
00072   (0 == strcmp(shpref_get(SHPREF_TRACK), "true") ? TRUE : FALSE) 
00073 
00074 /**
00075  * Permanently sets the @c SHPREF_TRACK option.
00076  * @param opt A zero to disable the option and a non-zero to enable.
00077  */
00078 #define shpref_set_track(opt) \
00079   (opt ? shpref_set(SHPREF_TRACK, "true") : shpref_set(SHPREF_TRACK, "false"))

00081 /**
00082  * Specifies whether to overlay the sharefs filesystem ontop of the work directory on the local filesystem.
00083  * @note Disable this option to prevent libshare from writing outside of the base directory.
00084  * @returns A zero (0) when disabled and a non-zero value when enabled.
00085  */
00086 #define shpref_overlay() \
